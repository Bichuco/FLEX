const SHEET_USERS = 'Usuarios';
const SHEET_ORDERS = 'Pedidos';

function doPost(e){

const data = JSON.parse(e.postData.contents);

switch(data.action){

case 'register':
return registerUser(data);

case 'login':
return loginUser(data);

case 'createOrder':
return createOrder(data);

case 'getOrders':
return getOrders(data);

}

}

function registerUser(data){

const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_USERS);
const rows = sheet.getDataRange().getValues();

for(let i=1;i<rows.length;i++){

if(rows[i][2] === data.email){
return output({success:false,message:'Usuario ya existe'});
}

}

sheet.appendRow([
new Date().getTime(),
data.name,
data.email,
data.password,
0,
'user'
]);

return output({success:true});

}

function loginUser(data){

const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_USERS);
const rows = sheet.getDataRange().getValues();

for(let i=1;i<rows.length;i++){

if(rows[i][2] === data.email && rows[i][3] === data.password){

return output({
success:true,
user:{
id:rows[i][0],
name:rows[i][1],
email:rows[i][2],
credits:rows[i][4],
role:rows[i][5]
}
});

}

}

return output({success:false,message:'Credenciales incorrectas'});

}

function createOrder(data){

const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_ORDERS);

sheet.appendRow([
new Date().getTime(),
data.user,
data.vehicle,
data.ecu,
data.original,
'',
'Pendiente',
new Date()
]);

return output({success:true});

}

function getOrders(data){

const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_ORDERS);
const rows = sheet.getDataRange().getValues();
const result = [];

for(let i=1;i<rows.length;i++){

if(rows[i][1] === data.user){
result.push(rows[i]);
}

}

return output({success:true,orders:result});

}

function output(data){

return ContentService
.createTextOutput(JSON.stringify(data))
.setMimeType(ContentService.MimeType.JSON);

}
