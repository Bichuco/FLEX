const SHEET_NAME = 'Proyectos';
const ROOT_FOLDER_ID = 'PON_AQUI_EL_ID_DE_LA_CARPETA_RAIZ';

function doGet() {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle('Gestión de proyectos de modificación de archivos');
}

function getSheet_() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
}

function listarProyectos() {
  const sheet = getSheet_();
  const data = sheet.getDataRange().getValues();
  data.shift(); // encabezados

  return data.map(row => ({
    id: row[0],
    matricula: row[1],
    vehiculo: row[2],
    tipo: row[3],
    ajustes: row[4],
    dtcs: row[5],
    comentarios: row[6],
    archivoOriginalId: row[7],
    carpetaId: row[8],
    fecha: row[9]
  }));
}

function getProyectoById(id) {
  const sheet = getSheet_();
  const data = sheet.getDataRange().getValues();
  data.shift(); // encabezados

  for (let row of data) {
    if (String(row[0]) === String(id)) {
      return {
        id: row[0],
        matricula: row[1],
        vehiculo: row[2],
        tipo: row[3],
        ajustes: row[4],
        dtcs: row[5],
        comentarios: row[6],
        archivoOriginalId: row[7],
        carpetaId: row[8],
        fecha: row[9]
      };
    }
  }
  throw new Error("Proyecto no encontrado");
}

function crearProyecto(datos, archivoOriginal) {
  const sheet = getSheet_();
  const newId = sheet.getLastRow(); // ID automático simple

  // Guardar valores únicos en listas auxiliares
  guardarValorUnico("Vehiculos", datos.vehiculo);
  guardarValorUnico("TiposCentralita", datos.tipo);

  const parentFolder = DriveApp.getFolderById(ROOT_FOLDER_ID);
  const projectFolder = parentFolder.createFolder("Proyecto_" + newId);

  const blob = Utilities.newBlob(
    archivoOriginal.bytes,
    archivoOriginal.mimeType,
    archivoOriginal.filename
  );
  const uploaded = projectFolder.createFile(blob);

  sheet.appendRow([
    newId,
    datos.matricula,
    datos.vehiculo,
    datos.tipo,
    datos.ajustes.join(", "),
    datos.dtcs,
    datos.comentarios,
    uploaded.getId(),
    projectFolder.getId(),
    new Date()
  ]);

  return { success: true, id: newId };
}

function listarArchivosProyecto(folderId) {
  const folder = DriveApp.getFolderById(folderId);
  const files = folder.getFiles();
  const result = [];

  while (files.hasNext()) {
    const f = files.next();
    result.push({
      id: f.getId(),
      nombre: f.getName(),
      url: f.getUrl()
    });
  }
  return result;
}

function subirArchivosModificados(folderId, archivos) {
  const folder = DriveApp.getFolderById(folderId);

  archivos.forEach(file => {
    const blob = Utilities.newBlob(file.bytes, file.mimeType, file.filename);
    folder.createFile(blob);
  });

  return { success: true };
}

// ---------- Listas auxiliares (vehículos y tipos de centralita) ----------

function guardarValorUnico(sheetName, value) {
  if (!value) return;

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.getRange(1, 1).setValue(sheetName === "Vehiculos" ? "Vehiculo" : "TipoCentralita");
  }

  const lastRow = sheet.getLastRow();
  const data = lastRow > 1
    ? sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat()
    : [];

  if (!data.includes(value)) {
    sheet.appendRow([value]);
  }
}

function obtenerLista(sheetName) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];

  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];

  const data = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat();
  return data.filter(v => v && v.toString().trim() !== "");
}

function obtenerVehiculos() {
  return obtenerLista("Vehiculos");
}

function obtenerTiposCentralita() {
  return obtenerLista("TiposCentralita");
}
